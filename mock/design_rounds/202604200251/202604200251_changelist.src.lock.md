/// One morsel's record slice.
#[derive(Copy, Clone, Debug, Default, Eq, PartialEq, Hash)]
pub struct MorselRange {
    pub start: u64,
    pub len: u32,
}

impl MorselRange {
    pub const fn new(start: u64, len: u32) -> Self { Self { start, len } }
    pub const fn end(&self) -> u64 { self.start + self.len as u64 }
    pub const fn is_empty(&self) -> bool { self.len == 0 }

### `src/dispatch/progress.rs` (~50 lines)

```rust
use core::sync::atomic::{AtomicUsize, Ordering};

/// Per-fiber monotonic record index.
///
/// Release store / Acquire load. Lock-free. Ships as a
/// `#[repr(transparent)]` newtype so downstream stores can lay out
/// a parallel array of ProgressCounters with no extra padding.
#[repr(transparent)]
#[derive(Debug, Default)]
pub struct ProgressCounter(AtomicUsize);

impl ProgressCounter {
    pub const fn new(start: usize) -> Self { Self(AtomicUsize::new(start)) }
    /// Release store. Publishes the new record index to any thread
    /// doing a later Acquire load.
    pub fn store(&self, value: usize) { self.0.store(value, Ordering::Release); }
    /// Acquire load. Pairs with a Release store from the writer.
    pub fn load(&self) -> usize { self.0.load(Ordering::Acquire) }
}
```

### `src/dispatch/fiber_dispatch.rs` (~55 lines)

```rust
use crate::plan::{FiberId, PhaseId};
use super::{MorselRange, SyncPoint, WuFn};

/// Per-fiber dispatch record. Pairs a monomorphised function
/// pointer with the metadata the engine needs to drive one fiber
/// through its morsel range under the right sync conditions.
///
/// `MAX_CORES` controls the sync-point array length: each fiber
/// has at most one SyncPoint per core that could run the phase
/// before it.
pub struct FiberDispatch<Ctx: 'static, const MAX_CORES: usize> {
    pub body: Option<WuFn<Ctx>>,
    pub fiber_id: FiberId,
    pub phase: PhaseId,
    pub morsel_range: MorselRange,
    pub sync_points: [SyncPoint; MAX_CORES],
    pub sync_point_count: u8,
}

impl<Ctx: 'static, const MAX_CORES: usize> FiberDispatch<Ctx, MAX_CORES> {
    pub const fn new() -> Self {
        Self {
            body: None,
            fiber_id: FiberId(0),
            phase: PhaseId(0),
            morsel_range: MorselRange { start: 0, len: 0 },
            sync_points: [SyncPoint { fiber_id: FiberId(0), min_records: 0 }; MAX_CORES],
            sync_point_count: 0,
        }
    }
}

impl<Ctx: 'static, const MAX_CORES: usize> Default for FiberDispatch<Ctx, MAX_CORES> {
    fn default() -> Self { Self::new() }
}
```

### `src/dispatch/core_dispatch.rs` (~70 lines)

```rust
use crate::plan::PhaseId;
use super::{FiberDispatch, MorselRange, SyncPoint};

/// Per-core compiled pipeline. Encodes the phases the core walks
/// through plus the fiber assignments inside each phase, plus the
/// morsel boundaries + sync points the core respects while doing
/// so. No dynamic dispatch at runtime: each slot is monomorphised.
pub struct CoreDispatch<Ctx: 'static, const MAX_FIBERS: usize> {
    pub fibers: [FiberDispatch<Ctx, MAX_FIBERS>; MAX_FIBERS],
    pub fiber_count: u16,
    pub phases: [PhaseId; MAX_FIBERS],
    pub phase_count: u8,
    pub morsel_boundaries: [MorselRange; MAX_FIBERS],
    pub boundary_count: u16,
    pub sync_points: [SyncPoint; MAX_FIBERS],
    pub sync_point_count: u16,
}

impl<Ctx: 'static, const MAX_FIBERS: usize> CoreDispatch<Ctx, MAX_FIBERS> {
    pub fn new() -> Self {
        Self {
            fibers: core::array::from_fn(|_| FiberDispatch::new()),
            fiber_count: 0,
            phases: [PhaseId(0); MAX_FIBERS],
            phase_count: 0,
            morsel_boundaries: [MorselRange { start: 0, len: 0 }; MAX_FIBERS],
            boundary_count: 0,
            sync_points: [SyncPoint { fiber_id: crate::plan::FiberId(0), min_records: 0 }; MAX_FIBERS],
            sync_point_count: 0,
        }
    }
}

impl<Ctx: 'static, const MAX_FIBERS: usize> Default for CoreDispatch<Ctx, MAX_FIBERS> {
    fn default() -> Self { Self::new() }
}
```

### `src/intrinsics/mod.rs` (~60 lines)

Zero-cost no-op shims on stable, gated upgrades behind the
`intrinsics-std` feature. Uses `core::hint` + `core::sync::atomic`
only — no `std::` path, no nightly-only intrinsics (those are a
BACKLOG entry).

```rust
//! Low-level engine intrinsics shim (domain 13).
//!
//! Portable, no-alloc, no-std. Ships as zero-cost stubs on
//! stable with optional upgrades under `intrinsics-std` (still
//! core-only but uses `core::hint::black_box` etc.).

use core::sync::atomic::{compiler_fence, Ordering};

pub fn prefetch_l1<T>(ptr: *const T) {
    let _ = ptr;
}

pub fn prefetch_l2<T>(ptr: *const T) {
    let _ = ptr;
}

pub fn compiler_fence_release() {
    compiler_fence(Ordering::Release);
}

pub fn compiler_fence_acquire() {
    compiler_fence(Ordering::Acquire);
}

#[cfg(feature = "intrinsics-std")]
pub fn noinline_barrier<T>(val: T) -> T {
    core::hint::black_box(val)
}

#[cfg(not(feature = "intrinsics-std"))]
pub fn noinline_barrier<T>(val: T) -> T {
    val
}
```

## Impl-time decisions

1. **`WuFn<Ctx> = fn(&Ctx)`** — matches the monomorphised form
   of `WorkUnit::execute(&self, &Ctx)`. The `&self` receiver gets
   closed over at monomorphisation time (5a3 stub doesn't need
   to model that machinery).
2. **`FiberDispatch` + `CoreDispatch` carry `Ctx: 'static`** — the
   dispatch record is monomorphised per-Ctx; different pipelines
   produce different instantiations. The `'static` bound matches
   `WorkUnit::Ctx`.
3. **`sync_points: [SyncPoint; MAX_CORES]`** on FiberDispatch —
   upper-bound on sync-point count; `sync_point_count: u8` holds
   the actual used count.
4. **`ProgressCounter` is `#[repr(transparent)]`** — lets downstream
   code do parallel-array layout with no extra padding. Release
   store + Acquire load (no RMW — cheapest atomic).
5. **Intrinsics: stable-only shims** — the task brief mentions a
   possible `core::intrinsics::prefetch_read_data` hook under an
   `intrinsics-std` feature, but `core::intrinsics` is nightly and
   this crate targets stable; keeping the shim no-op on stable and
   only pulling in `core::hint::black_box` under `intrinsics-std`.
   A real `nightly-intrinsics` feature lives in BACKLOG.
6. **Progress counter is not tested for concurrency** — unit tests
   cover the single-threaded contract; loom/miri is a BACKLOG
   entry.
7. **`CoreDispatch::new` uses `core::array::from_fn`** — the
   FiberDispatch field is not Copy (has a function pointer
   Option), so `[FiberDispatch::new(); MAX_FIBERS]` can't be a
   const array literal. `from_fn` works in non-const context;
   const-construction is a follow-up.
